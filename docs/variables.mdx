export const filterVariables = (variables, query) => {
  if (!query) {
    return variables;
  }

  const normalizedQuery = query.toLowerCase();
  return variables.filter((value) => {
    const description = getDescription(value).toLowerCase();
    return value.toLowerCase().includes(normalizedQuery) || description.includes(normalizedQuery);
  });
};

export const getAlertEntriesForPlatform = (platform) =>
  Object.entries(LumiaAlertConfigs)
    .filter(([, config]) => config.connection === platform)
    .sort((a, b) => titleCase(a[0]).localeCompare(titleCase(b[0])));

    alert,

    ...(config.acceptedVariables || []),
    ...(config.quickActions || []).map((quickAction) => quickAction.label),
  ]
    .filter(Boolean)
    .join(" ")
    .toLowerCase();

export const getAlertFieldExamples = (config, field) => {
  const examples = [];
  const seen = new Set();

  (config.quickActions || []).forEach((quickAction) => {
    const value = quickAction?.extraSettings?.[field];
    if (value === undefined || value === null || value === "") {
      return;
    }

    const serialized = serializeExampleValue(value);
    if (seen.has(serialized)) {
      return;
    }

    seen.add(serialized);
    examples.push(serialized);
  });

  (config.inputFields || []).forEach((inputField) => {
    if (inputField.variableField !== field) {
      return;
    }

    if (inputField.default === undefined || inputField.default === null || inputField.default === "") {
      return;
    }

    const serialized = serializeExampleValue(inputField.default);
    if (seen.has(serialized)) {
      return;
    }

    seen.add(serialized);
    examples.push(serialized);
  });

  return examples.slice(0, 6);
};

export const getAlertExamplePayload = (alert, config, exampleIndex = 0) => {
  const exampleAction = config.quickActions?.[exampleIndex] || config.quickActions?.[0];
  const extraSettings = {};
  const dynamic = {};

  (config.inputFields || []).forEach((inputField) => {
    if (inputField.variableField && inputField.default !== undefined && inputField.default !== null && inputField.default !== "") {
      extraSettings[inputField.variableField] = inputField.default;
    }

    if (inputField.dynamicField && inputField.default !== undefined && inputField.default !== null && inputField.default !== "") {
      dynamic[inputField.dynamicField] = inputField.default;
    }
  });

  if (exampleAction?.extraSettings) {
    Object.assign(extraSettings, exampleAction.extraSettings);
  }

  if (exampleAction?.dynamic) {
    Object.assign(dynamic, exampleAction.dynamic);
  }

  const payload = {
    alert,
    extraSettings,
    fromLumia: true,
  };

  if (Object.keys(dynamic).length > 0) {
    payload.dynamic = dynamic;
  }

  return payload;
};

export const formatPayloadValue = (value) => {
  if (typeof value === "string") {
    return `"${value}"`;
  }

  if (value === null) {
    return "null";
  }

  if (typeof value === "undefined") {
    return "undefined";
  }

  return String(value);
};

export const PayloadTreeNode = ({ name, value, depth = 0 }) => {
  const isObject = value && typeof value === "object" && !Array.isArray(value);
  const isArray = Array.isArray(value);

  if (!isObject && !isArray) {
    return (
      <div style={{ marginLeft: `${depth * 1.1}rem`, marginBottom: "0.35rem" }}>
        <code>{name}</code>: <code>{formatPayloadValue(value)}</code>
      </div>
    );
  }

  const entries = isArray ? value.map((item, index) => [index, item]) : Object.entries(value);

  return (
    <details open={depth < 2} style={{ marginLeft: `${depth * 1.1}rem`, marginBottom: "0.5rem" }}>
      <summary style={{ cursor: "pointer" }}>
        <code>{name}</code>
        {" "}
        <span style={{ color: "var(--ifm-color-emphasis-700)" }}>
          {isArray ? `[${entries.length}]` : `{${entries.length}}`}
        </span>
      </summary>
      <div
        style={{
          marginTop: "0.5rem",
          paddingLeft: "0.85rem",
          borderLeft: "1px solid var(--ifm-toc-border-color)",
        }}
      >
        {entries.map(([key, childValue]) => (
          <PayloadTreeNode key={String(key)} name={String(key)} value={childValue} depth={depth + 1} />
        ))}
      </div>
    </details>
  );
};

export const AlertExplorer = ({ platform, alerts }) => {

    if (!alertQuery) {
      return true;
    }

    return getAlertSearchText(alert, config).includes(alertQuery.toLowerCase());
  });

    if (!filteredAlerts.some(([alert]) => alert === selectedAlert)) {
      setSelectedAlert(filteredAlerts[0]?.[0] || "");
      setSelectedExampleIndex(0);
    }
  }, [filteredAlerts, selectedAlert]);

  const activeEntry = filteredAlerts.find(([alert]) => alert === selectedAlert) || filteredAlerts[0];

    return null;

      <p style={{ marginTop: 0 }}>
        Search an alert and inspect a live example of what
        {" "}
        <code>Overlay.on('alert')</code>

        {" "}
        <code>data.extraSettings</code>.
      </p>
      <label style={{ display: "block", fontWeight: 600, marginBottom: "0.5rem" }}>
        Search alerts on {titleCase(platform)}

        type="search"

export const SearchablePlatformVariables = ({ platform, cfg }) => {
  const [query, setQuery] = React.useState("");

  const filteredGlobals = filterVariables(cfg.variables || [], query);
  const filteredChat = Object.entries(cfg.chat || {})
    .sort(([a], [b]) => {
      const aIs = chatCommandTypes.includes(a);
      const bIs = chatCommandTypes.includes(b);
      if (aIs !== bIs) {
        return aIs ? -1 : 1;
      }
      return a.localeCompare(b);
    })
    .map(([type, variables]) => [type, filterVariables(variables || [], query)])
    .filter(([, variables]) => variables.length > 0 || !query);

  const alertConfigEntries = getAlertEntriesForPlatform(platform);

  const hasMatches =
    filteredGlobals.length > 0 ||
    filteredChat.length > 0;

  return (
    <>
      <div style={boxStyles}>
        <div style={chipStyles}>{titleCase(platform)}</div>
        <p style={{ marginTop: 0, marginBottom: "0.75rem" }}>
          Variables only work when the event provides that data. Global variables are broadly available. Chat and alert variables only work in those specific contexts.
        </p>
        <label style={{ display: "block", fontWeight: 600, marginBottom: "0.5rem" }}>
          Search variables on this platform
        </label>
        <input
          type="search"
          value={query}
          onChange={(event) => setQuery(event.target.value)}
          placeholder="Search by variable name or meaning"
          style={{
            width: "100%",
            maxWidth: "520px",
            padding: "0.75rem 0.9rem",
            borderRadius: "10px",
            border: "1px solid var(--ifm-toc-border-color)",
            background: "var(--ifm-background-surface-color)",
          }}
        />
      </div>

      {cfg.variables?.length > 0 && (
        <section style={boxStyles}>
          <div style={chipStyles}>Available everywhere</div>
          <p style={{ marginTop: 0 }}>
            These are the safest variables to use across the selected platform. This table includes sample output values.
          </p>
          <VariableTable
            variables={filteredGlobals}
            emptyLabel={`No global variables matched "${query}".`}
            getExamples={(value) => getVariableExamples(platform, value)}
          />
        </section>
      )}

      {filteredChat.length > 0 && (
        <section style={boxStyles}>
          <div style={chipStyles}>Chat and commands</div>
          <p style={{ marginTop: 0 }}>
            Use these when building chat responses, command replies, redeems, or chatbot actions.
          </p>
          {filteredChat.map(([type, variables]) => (
            <details key={type} open={!query} style={{ marginBottom: "1rem" }}>
              <summary style={{ cursor: "pointer", fontWeight: 700, marginBottom: "0.75rem" }}>
                {getTypeLabel(type)}
                {" "}
                <span style={{ color: "var(--ifm-color-emphasis-700)", fontWeight: 500 }}>
                  ({variables.length})
                </span>
              </summary>
              <VariableTable
                variables={variables}
                emptyLabel={`No variables matched "${query}" in ${getTypeLabel(type)}.`}
              />
            </details>
          ))}
        </section>
      )}

      <AlertExplorer platform={platform} alerts={alertConfigEntries} />

      {!hasMatches && query && (
        <div style={boxStyles}>
          <div style={chipStyles}>No matches</div>
          <p style={{ margin: 0 }}>
            No variables matched "{query}" on {titleCase(platform)}. Try a broader term like
            {" "}
            <code>username</code>, <code>amount</code>, or <code>message</code>.
          </p>
        </div>
      )}
    </>
  );
};
