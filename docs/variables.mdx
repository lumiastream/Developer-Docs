import React from "react";
import Link from "@docusaurus/Link";
import { AllVariables } from "@lumiastream/lumia-types";
import { enTrans } from "@lumiastream/lumia-translations";
import Tabs from "@theme/Tabs";
import TabItem from "@theme/TabItem";

export const chatCommandTypes = [
  "command",
  "chatbotCommand",
  "twitchPoints",
  "kickPoints",
  "trovoSpells",
  "twitchExtensions",
  "chatmatch",
];

export const slugify = (s) =>
  s
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, "-")
    .replace(/^-+|-+$/g, "");

# Platform Variables

<Tabs queryString="platform">
  {Object.entries(AllVariables).map(([platform, cfg]) => (
    <TabItem key={platform} value={platform} label={platform.charAt(0).toUpperCase() + platform.slice(1)}>
      
      {/* Global Variables */}
      {cfg.variables?.length > 0 && (
        <>
          <h3 style={{ color: "#701c34", textTransform: "capitalize" }}>
            {platform === "lumiastream" ? "Global Variables" : `${platform} Global Variables`}
          </h3>
          <ul
            style={{
              padding: "10px",
              borderRadius: "4px",
              listStyle: "disc inside",
            }}
          >
            {cfg.variables.map((v) => (
              <li key={v} style={{ padding: "4px 0" }}>
                <code>{v}</code>
                {enTrans?.variables?.[v] && (
                  <div style={{ marginLeft: "8px", color: "#555" }}>
                    {enTrans.variables[v]}
                  </div>
                )}
              </li>
            ))}
          </ul>
        </>
      )}

      {/* Chat Command Types */}
      {cfg.chat && Object.keys(cfg.chat).length > 0 && (
        <>
          {Object.entries(cfg.chat)
            .sort(([a], [b]) => {
              const aIs = chatCommandTypes.includes(a);
              const bIs = chatCommandTypes.includes(b);
              if (aIs !== bIs) return aIs ? -1 : 1;
              return a.localeCompare(b);
            })
            .map(([type, vars]) => {
              const label = chatCommandTypes.includes(type)
                ? `${type} [Chat Type]`
                : `${type} [Chatbot Command Type]`;
              return (
                <React.Fragment key={type}>
                  <h3 style={{ color: "#701c34", textTransform: "capitalize" }}>
                    {label}
                  </h3>
                  {vars.length > 0 ? (
                    <ul
                      style={{
                        padding: "10px",
                        borderRadius: "4px",
                        listStyle: "circle inside",
                      }}
                    >
                      {vars.map((a) => (
                        <li key={a} style={{ padding: "4px 0" }}>
                          <code>{a}</code>
                          {enTrans?.variables?.[a] && (
                            <div style={{ marginLeft: "8px", color: "#555" }}>
                              {enTrans.variables[a]}
                            </div>
                          )}
                        </li>
                      ))}
                    </ul>
                  ) : (
                    <em>none</em>
                  )}
                </React.Fragment>
              );
            })}
        </>
      )}

      {/* Alerts */}
      {cfg.alerts && Object.keys(cfg.alerts).length > 0 && (
        <>
          <h3 style={{ color: "#701c34", textTransform: "capitalize" }}>
            {platform} Alerts Variables
          </h3>
          <ul
            style={{
              padding: "10px",
              borderRadius: "4px",
              listStyle: "square inside",
            }}
          >
            {Object.entries(cfg.alerts).map(([alert, vars]) => (
              <li key={alert} style={{ padding: "4px 0" }}>
                <strong style={{ textTransform: "capitalize" }}>
                  {alert}
                </strong>
                :{" "}
                {vars.length > 0 ? (
                  <ul
                    style={{
                      marginLeft: "20px",
                      padding: "10px",
                      borderRadius: "4px",
                      listStyle: "disc inside",
                    }}
                  >
                    {vars.map((a) => (
                      <li key={a} style={{ padding: "4px 0" }}>
                        <code>{a}</code>
                        {enTrans?.variables?.[a] && (
                          <div style={{ marginLeft: "8px", color: "#555" }}>
                            {enTrans.variables[a]}
                          </div>
                        )}
                      </li>
                    ))}
                  </ul>
                ) : (
                  <em>none</em>
                )}
              </li>
            ))}
          </ul>
        </>
      )}
    </TabItem>

))}

</Tabs>
