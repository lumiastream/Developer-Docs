export const chatEventExample = `Overlay.on('chat', (data) => {
  const name = data.displayname || data.username || 'Unknown';
  const platform = data.origin;

  console.log(\`[\${platform}] \${name}: \${data.message}\`);

  if (data.reply) {
    console.log(\`Replying to \${data.reply.username}: \${data.reply.body}\`);
  }

  if (data.isPowerup) {
    console.log(\`Power-up: \${data.powerupType} / \${data.powerupName}\`);
  }
});`;

export const alertEventExample = `Overlay.on('alert', (data) => {
  switch (data.alert) {
    case LumiaAlertValues.TWITCH_FOLLOWER:
      console.log(\`New follower: \${data.extraSettings.username}\`);
      break;

    case LumiaAlertValues.YOUTUBE_SUPERCHAT:
      console.log(
        \`\${data.extraSettings.username} sent \${data.extraSettings.currencySymbol}\${data.extraSettings.amount}\`
      );
      break;

    default:
      console.log('Alert type:', data.alert);
      console.log('Returned fields:', data.extraSettings);
  }
});`;
