import React from "react";
import { AllVariables } from "@lumiastream/lumia-types";
import Tabs from "@theme/Tabs";
import TabItem from "@theme/TabItem";
import {
  chatCommandTypes,
  titleCase,
  getVariableName,
  getVariableList,
  getDescription,
  getTypeLabel,
  tableStyles,
  tableWrapperStyles,
  thStyles,
  tdStyles,
  boxStyles,
  chipStyles,
  exampleChipStyles,
  filterVariables,
  getAlertEntriesForPlatform,
  getVariableExamples,
  VariableTable,
  AlertExplorer,
} from "@site/src/components/AlertExplorer";

export const SearchablePlatformVariables = ({ platform, cfg }) => {
  const [query, setQuery] = React.useState("");

  const globals = getVariableList(cfg.variables);
  const filteredGlobals = filterVariables(globals, query);
  const filteredChat = Object.entries(cfg.chat || {})
    .sort(([a], [b]) => {
      const aIs = chatCommandTypes.includes(a);
      const bIs = chatCommandTypes.includes(b);
      if (aIs !== bIs) return aIs ? -1 : 1;
      return a.localeCompare(b);
    })
    .map(([type, variables]) => [type, filterVariables(variables, query)])
    .filter(([, variables]) => variables.length > 0 || !query);

  const alertConfigEntries = getAlertEntriesForPlatform(platform);
  const hasMatches = filteredGlobals.length > 0 || filteredChat.length > 0;

  return (
    <>
      <div style={boxStyles}>
        <div style={chipStyles}>{titleCase(platform)}</div>
        <p style={{ marginTop: 0, marginBottom: "0.75rem" }}>
          Variables only work when the event provides that data. Global variables are broadly available. Chat and alert variables only work in those specific contexts.
        </p>
        <label style={{ display: "block", fontWeight: 600, marginBottom: "0.5rem" }}>
          Search variables on this platform
        </label>
        <input
          type="search"
          value={query}
          onChange={(event) => setQuery(event.target.value)}
          placeholder="Search by variable name or meaning"
          style={{
            width: "100%",
            maxWidth: "520px",
            padding: "0.75rem 0.9rem",
            borderRadius: "10px",
            border: "1px solid var(--ifm-toc-border-color)",
            background: "var(--ifm-background-surface-color)",
          }}
        />
      </div>

      {globals.length > 0 && (
        <section style={boxStyles}>
          <div style={chipStyles}>Available everywhere</div>
          <p style={{ marginTop: 0 }}>
            These are the safest variables to use across the selected platform. This table includes sample output values.
          </p>
          <VariableTable
            variables={filteredGlobals}
            emptyLabel={`No global variables matched "${query}".`}
            getExamples={(value) => getVariableExamples(platform, value)}
          />
        </section>
      )}

      {filteredChat.length > 0 && (
        <section style={boxStyles}>
          <div style={chipStyles}>Chat and commands</div>
          <p style={{ marginTop: 0 }}>
            Use these when building chat responses, command replies, redeems, or chatbot actions.
          </p>
          {filteredChat.map(([type, variables]) => (
            <details key={type} open={!query} style={{ marginBottom: "1rem" }}>
              <summary style={{ cursor: "pointer", fontWeight: 700, marginBottom: "0.75rem" }}>
                {getTypeLabel(type)}
                {" "}
                <span style={{ color: "var(--ifm-color-emphasis-700)", fontWeight: 500 }}>
                  ({variables.length})
                </span>
              </summary>
              <VariableTable
                variables={variables}
                emptyLabel={`No variables matched "${query}" in ${getTypeLabel(type)}.`}
              />
            </details>
          ))}
        </section>
      )}

      <AlertExplorer platform={platform} alerts={alertConfigEntries} />

      {!hasMatches && query && (
        <div style={boxStyles}>
          <div style={chipStyles}>No matches</div>
          <p style={{ margin: 0 }}>
            No variables matched "{query}" on {titleCase(platform)}. Try a broader term like{" "}
            <code>username</code>, <code>amount</code>, or <code>message</code>.
          </p>
        </div>
      )}
    </>
  );
};

# Platform Variables

Use variables as placeholders inside Lumia messages, alerts, and overlays. Write them like <code>{"{{username}}"}</code> or <code>{"{{amount}}"}</code>, and Lumia replaces them with live data when the event runs.

## Alert Explorer

If you are building a custom overlay, open a platform tab and use the alert explorer to inspect the exact alert name, the `data.extraSettings.*` fields it returns, and the matching template variables. For the full `AlertEvent` type, see [Overlay Type Definitions](/docs/custom-overlays/types#alertevent).
